    <script>
        class AnnaSathiOS {
            constructor() {
                this.appLanguageContext = "HI";
                this.currentTheme = "light";
                this.statesRegistry = this.getFallbackStatesRegistry();
                this.diagnosticDB = this.getProductionDiagnosticDatabase();
                this.languageMatrixText = this.getLanguageMatrix();
            }

            async init() {
                this.initErrorBoundary();
                this.registerServiceWorker();
                await this.fetchAndPopulateDynamicData();
                this.bindEvents();
                this.initDefaults();
            }

            initErrorBoundary() {
                window.addEventListener('error', (e) => {
                    console.error(`[AnnaSathi Framework Execution Exception]: ${e.message}`);
                    return true;
                });
            }

            registerServiceWorker() {
                if ('serviceWorker' in navigator) {
                    const swCode = `
                        const CACHE_NAME = 'annasathi-enterprise-v10';
                        const STATIC_ASSETS = ['/', 'index.html'];
                        self.addEventListener('install', (e) => e.waitUntil(caches.open(CACHE_NAME).then((c) => c.addAll(STATIC_ASSETS))));
                        self.addEventListener('fetch', (e) => e.respondWith(caches.match(e.request).then((res) => res || fetch(e.request))));
                    `;
                    const blob = new Blob([swCode], { type: 'application/javascript' });
                    navigator.serviceWorker.register(URL.createObjectURL(blob))
                    .catch(err => console.error("PWA Subsystem Initialization Fault:", err));
                }
            }

            async fetchAndPopulateDynamicData() {
                const stateSel = document.getElementById('stateSelector');
                const cameraCropSel = document.getElementById('cameraCropContext');
                const marketCropSel = document.getElementById('marketCrop');

                try {
                    const regionsResponse = await fetch('/api/v1/regions');
                    const statesData = regionsResponse.ok ? await regionsResponse.json() : this.statesRegistry;
                    
                    stateSel.innerHTML = '<option value="">-- Choose Region Node (36 States/UTs) --</option>';
                    statesData.forEach(node => {
                        const opt = document.createElement('option');
                        opt.value = node.id; opt.innerText = node.name;
                        stateSel.appendChild(opt);
                    });

                    const cropsResponse = await fetch('/api/v1/crops');
                    const cropsData = cropsResponse.ok ? await cropsResponse.json() : this.getFallbackCropTaxonomy();

                    cameraCropSel.innerHTML = '';
                    marketCropSel.innerHTML = '';
                    
                    Object.keys(cropsData).forEach(category => {
                        const groupCam = document.createElement('optgroup');
                        const groupMkt = document.createElement('optgroup');
                        groupCam.label = category; groupMkt.label = category;

                        cropsData[category].forEach(crop => {
                            const optCam = document.createElement('option');
                            const optMkt = document.createElement('option');
                            optCam.value = crop.id; optCam.innerText = crop.name;
                            optMkt.value = crop.basePrice;
                            optMkt.innerText = `${crop.name} (Base: ₹${crop.basePrice}/Qtl)`;
                            
                            groupCam.appendChild(optCam);
                            groupMkt.appendChild(optMkt);
                        });
                        cameraCropSel.appendChild(groupCam);
                        marketCropSel.appendChild(groupMkt);
                    });
                } catch (error) {
                    console.warn("REST API pipeline offline fallback matrix engaged.");
                }
            }

            bindEvents() {
                document.getElementById('btnToggleLanguage').addEventListener('click', () => this.toggleAppLanguage());
                document.getElementById('btnToggleTheme').addEventListener('click', () => this.toggleTheme());
                document.getElementById('btnSyncGPS').addEventListener('click', () => this.syncGPSHardwareEngine());
                document.getElementById('masterSearch').addEventListener('keyup', () => this.filterMasterProblems());
                document.getElementById('masterProblemSelect').addEventListener('change', () => this.runUltimateActionEngine());
                document.getElementById('ultimateTriggerBtn').addEventListener('click', () => this.triggerAgriOSProtocol());
                document.getElementById('btnLaunchCamera').addEventListener('click', () => document.getElementById('cameraHardwareTrigger').click());
                document.getElementById('cameraHardwareTrigger').addEventListener('change', (e) => this.processCapturedPhoto(e));
                document.getElementById('btnPrintReport').addEventListener('click', () => window.print());
                document.getElementById('stateSelector').addEventListener('change', (e) => this.updateDistrictHubs(e.target.value));
                document.getElementById('enterpriseRoleSelector').addEventListener('change', (e) => this.executeRoleIsolationPipeline(e.target.value));

                document.getElementById('marketCrop').addEventListener('change', () => {
                    document.getElementById('marketPrice').value = document.getElementById('marketCrop').value;
                    this.runMandiCalcs();
                });
                document.getElementById('marketQty').addEventListener('input', () => this.runMandiCalcs());
                document.getElementById('marketPrice').addEventListener('input', () => this.runMandiCalcs());
                document.getElementById('mandiTradingForm').addEventListener('submit', (e) => this.executeSecureTradeEscrow(e));

                document.querySelectorAll('.tab-btn, .nav-item').forEach(btn => {
                    btn.addEventListener('click', (e) => this.switchTab(e.currentTarget.getAttribute('data-tab')));
                });
            }

            switchTab(tabId) {
                document.querySelectorAll('.tab-content').forEach(el => el.classList.remove('active'));
                document.getElementById('tab-content-' + tabId).classList.add('active');
            }

            runMandiCalcs() {

            processCapturedPhoto(event) {
                const loader = document.getElementById('cameraScanLoader');
                const resultBox = document.getElementById('cameraScanResult');
                const cropContext = document.getElementById('cameraCropContext').value;
                
                const dTitle = document.getElementById('cameraDetectedDiseaseTitle');
                const dSymptoms = document.getElementById('cameraDetectedSymptoms');
                const dSolutions = document.getElementById('cameraDetectedSolutions');
                const dMedicine = document.getElementById('cameraDetectedMedicine');

                if(event.target.files && event.target.files[0]) {
                    resultBox.classList.add('hidden'); loader.classList.remove('hidden');

                    const file = event.target.files[0];
                    const img = new Image();
                    img.src = URL.createObjectURL(file);
                    img.onload = () => {
                        const canvas = document.createElement('canvas');
                        const ctx = canvas.getContext('2d');
                        const maxDimension = 512;
                        let width = img.width; let height = img.height;

                        if (width > height) {
                            if (width > maxDimension) { height *= maxDimension / width; width = maxDimension; }
                        } else {
                            if (height > maxDimension) { width *= maxDimension / height; height = maxDimension; }
                        }
                        canvas.width = width; canvas.height = height;
                        ctx.drawImage(img, 0, 0, width, height);
                        
                        if(window.tf) {
                            window.tf.tidy(() => {
                                const tensor = window.tf.browser.fromPixels(canvas);
                                console.log("TensorFlow.js edge input scale verification verified:", tensor.shape);
                            });
                        }
                    };

                    setTimeout(() => {
                        loader.classList.add('hidden'); resultBox.classList.remove('hidden');
                        const matchedOutput = this.diagnosticDB[cropContext] ? this.diagnosticDB[cropContext][this.appLanguageContext] : this.diagnosticDB['paddy'][this.appLanguageContext];
                        
                        dTitle.innerText = matchedOutput.t; dSymptoms.innerHTML = matchedOutput.s;
                        dSolutions.innerHTML = matchedOutput.g; dMedicine.innerHTML = matchedOutput.m;
                    }, 2500);
                }
            }
